# CNC Winder Dashboard Configuration
# Add to Mainsail settings for CNC-specific interface

# Custom Macros Panel (add these to Mainsail dashboard)
# These will appear as touch buttons on your interface

WIND_COIL TURNS=5000 SPEED=1000    # Wind 5000 turns at speed 1000
WIND_COIL TURNS=10000 SPEED=800    # Wind 10000 turns at speed 800  
WIND_COIL TURNS=25000 SPEED=600    # Wind 25000 turns at speed 600

HOME_ALL                            # Home all motors
EMERGENCY_STOP                      # Emergency stop

# Status Display Items (add to dashboard)

# CNC Winder Operation:
